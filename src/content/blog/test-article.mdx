# Live shader

```glsl live
#ifdef GL_ES
precision mediump float;
#endif

uniform vec2 u_resolution;
uniform float u_time;

void main(){

vec2 uv = gl_FragCoord.xy / u_resolution;

vec3 col = 0.5 + 0.5 * sin(u_time + uv.xyx + vec3(0.0,2.0,4.0));

gl_FragColor = vec4(col,1.0);

}
```
